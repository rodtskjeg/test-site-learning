import java.util.Objects;
      class Rational {
      private int numerator;
      private int denominator;
      public Rational(int numerator, int denominator) {
      if (denominator == 0) {
      throw new IllegalArgumentException("Denominator cannot be zero");
      }
      int gcd = gcd(numerator, denominator);
      this.numerator = numerator / gcd;
      this.denominator = denominator / gcd;
      if (this.denominator < 0) { // keep the denominator positive
      this.numerator = -this.numerator;
      this.denominator = -this.denominator;
      }
      }
      private int gcd(int a, int b) {
      if (b == 0) return a;
      return gcd(b, a % b);
      }
      @Override
      public boolean equals(Object o) {
      if (this == o) return true;
      if (o == null || getClass() != o.getClass()) return false;
      Rational rational = (Rational) o;
      return (double)numerator / denominator == (double)rational.numerator / rational.denominator;
      }
      @Override
      public int hashCode() {
      return Objects.hash((double)numerator / denominator);
      }
      @Override
      public String toString() {
      return numerator + "/" + denominator;
      }
      }
      
      
      import java.util.Objects;
      
      class RatRange {
        private Rational start;
        private Rational end;
      
        public RatRange(Rational start, Rational end) {
      this.start = start;
      this.end = end;
        }
      
        @Override
        public boolean equals(Object o) {
      if (this == o) return true;
      if (o == null || getClass() != o.getClass()) return false;
      RatRange ratRange = (RatRange) o;
      return start.equals(ratRange.start) && end.equals(ratRange.end);
        }
      
        @Override
        public int hashCode() {
      return Objects.hash((double) start.hashCode() / end.hashCode()); // Import added here
        }
      
        @Override
        public String toString() {
      return "RatRange{" + "start=" + start + ", end=" + end + '}';
        }
      }
      
      
      public class Main {
      public static void main(String[] args) {
      Rational r1 = new Rational(1, 2);
      Rational r2 = new Rational(2, 4);
      Rational r3 = new Rational(3, 4);
      RatRange range1 = new RatRange(r1, r2);
      RatRange range2 = new RatRange(r2, r3);
      RatRange range3 = new RatRange(r1, r3);
      System.out.println("RatRange range1 equals range2: " + range1.equals(range2)); // should print true
      System.out.println("RatRange range1 equals range3: " + range1.equals(range3)); // should print false
      System.out.println("RatRange range1: " + range1);
      System.out.println("RatRange range2: " + range2);
      System.out.println("RatRange range3: " + range3);
      }
      }
